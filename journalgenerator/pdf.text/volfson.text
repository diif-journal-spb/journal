    \textbf{ЛОКАЛЬНАЯ ИДЕНТИФИЦИРУЕМОСТЬ СИСТЕМ С ПЕРЕМЕННЫМ ПАРАМЕТРОМ}

\end{center}

    Рассмотрим модель, описываемую системой обыкновенных дифференциальных уравнений:

    \begin{equation}
    \label{first_eq}
    \dot x = f(t, x, p),
    \end{equation}
    где $x\in \mathbb{R}^n$, $p\in \mathbb{R}^m$.

    \medskip
    Мы предполагаем, что вектор-функция $f$, стоящая в правой части системы (\ref{first_eq}), непрерывна по совокупности переменных $(t,x,p)$ и непрерывно дифференцируема по переменным $(x,p)$ в области $D$ пространства $\mathbb{R}^{1+n+m}$.

    \medskip
    В изучаемой постановке задачи параметр $p$ является непрерывной функцией от $t$, принадлежащей некоторому классу функций $\EuScript P$, определенных на фиксированном промежутке $[0,T]$. Введем обозначение $\{p\}=\{p(t):t\in[0;T]\}$.

    \medskip
    Фиксируем вектор $x_0 \in \mathbb{R}^n$ и обозначим через $x(t, \{p\})$ решение задачи Коши с начальными данными $(0, x_0)$ для системы (\ref{first_eq}), в которой зафиксирован параметр-функция $p \in \EuScript P$.

    \medskip
    Мы учитываем, что значение $x(t_0, \{p\})$ при $t_0 \in [0,T]$ зависит лишь от значений $p(t)$ при $t \in [0,t_0]$.

    \medskip
    Наше основное предположение о классе $\EuScript P$ таково: для любой функции $p \in \EuScript P$ решение $x(t,\{p\})$ единственно и определено на всем промежутке $[0,T]$.

    \medskip
    Для любой функции $q(t)$, $t \in [0,T]$, мы обозначаем

    $$
        \|q\| = \sup\limits_{t \in [0,T]} |q(t)|.
    $$

    \medskip
    \textbf{Определение 1.} Пусть $p_1$, $p_2 \in \EuScript P$. Будем говорить, что пара $(p_1, p_2)$ различима по наблюдению решения $x(t, \{p\})$ в точке $T$, если

    $$
    x(T, \{p_1\}) \ne x(T, \{p_2\}).
    $$

    \medskip
    В дальнейшем, для краткости будем говорить просто "пара $(p_1, p_2)$ различима". Отметим, что вектор $x_0$ предполагается фиксированным для всех рассматриваемых решений $x(t, \{p\})$.

    \medskip
    \textbf{Определение 2.} Система (\ref{first_eq}) называется локально идентифицируемой при $p_0 \in \EuScript P$, если существует такое $\epsilon > 0$, что если $p \in \EuScript P$ и $0<\|p_0-p\|<\epsilon$, то пара $(p_0, p)$  различима.

    \medskip
    Постановка вопроса о локальной идентифицируемости в задаче об идентифицируемости решения по наблюдениям на концах промежутка была исследована в монографии [1]. Исследование проводилось в том случае, когда параметр-функция $p$ -- постоянная или кусочно-постоянная на отрезке $[0, T]$. Основным отличием данной постановки задачи от постановки задачи в [1] является то, что мы не предполагаем постоянство или кусочное постоянство параметр-функции $p$.

    \medskip
    Пусть $\EuScript K$ - некоторый класс функций, заданных на $[0,T]$.

    \medskip
    \textbf{Определение 3.} Фиксируем $p_0 \in \EuScript P$. Будем говорить, что семейство $\EuScript P$ нормированно отделено от $\EuScript K$ в $p_0$, если из любой последовательности элементов $p_k \in \EuScript P$, обладающей свойствами $\|p_k - p_0 \| > 0$ и $\|p_k - p_0 \| \rightarrow 0, k\rightarrow\infty$, можно выбрать такую подпоследовательность $p_{k_l}$, что последовательность $\frac{p_{k_l} - p_0}{\|p_{k_l} - p_0\|}$ равномерно сходится на $[0,T]$ к некоторой функции, не принадлежащей классу $\EuScript K$.

    \medskip
    Фиксируем $p_0 \in \EuScript P$.

    Предположим, что
    \begin{equation}
    \label{first_equ}
    \{(t,x(t,\{p_0\}),p_0(t)):t\in [0,T]\} \subset D.
    \end{equation}

    Пусть $Y(t)$ - фундаментальная матрица линейной системы

    \begin{equation}
    \label{second_eq}
    \dot y = \frac{\partial f(t,x(t,\{p_0\}),p_0)}{\partial x} y,
    \end{equation}
    удовлетворяющая начальному условию

    $$
    Y(0) = E,
    $$
    где $E$ - единичная матрица размера $n \times n$.

    \medskip

    Рассмотрим линейный функционал
    $$
    \Psi:\EuScript P\rightarrow \mathbb{R}^n,
    $$
    заданный формулой

    \begin{equation}
    \label{third_eq}
    \Psi(p) = \int\limits_0^T Y^{-1} (s) \cdot \frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))p(s)ds,
    \end{equation}
    и обозначим через $\EuScript K$ ядро функционала $\Psi$.

    \medskip
    Основной результат статьи таков.

    \medskip
    \textbf{Теорема.} Если семейство $\EuScript P$ нормированно отделено от $\EuScript K$ в $p_0$, то система (\ref{first_eq}) локально идентифицируема при параметр-функции $p_0$.

    \medskip
    \textbf{Доказательство.}

    \medskip
    Обозначим малое изменение параметра $p_0$ через $\Delta p$:

    $$
    p(t) = p_0(t) + \Delta p(t).
    $$


    \medskip
    Также введем обозначение:

    $$
    \Delta x(t) = x(t,\{p\}) - x(t,\{p_0\}).
    $$

    Докажем, что для $\Delta x(T)$ имеет место следующее представление:


    \begin{equation}
    \label{fourth_eq}
    \Delta x(T) = Y(T) \cdot \Psi(\Delta p) + G(\Delta p),
    \end{equation}
    где
    \begin{equation}
    \label{fourth_equ}
    \frac{|G(\Delta p)|}{\|\Delta p\|} \rightarrow 0 \mbox{ при } \|\Delta p\| \rightarrow 0.
    \end{equation}

    \medskip
    Подставим $x(t,\{p_0\})$ в (\ref{first_eq}) и проинтегрируем результат на промежутке $[0,t]$:

    \begin{equation}
    \label{fifth_eq}
    x(t, \{p_0\}) = x_0 + \int\limits_0^t f(s, x(s, \{p_0\}), p_0(s))ds.
    \end{equation}

    Аналогично, верно равенство
    $$
    x(t, \{p\}) = x_0 + \int\limits_0^t f(s, x(s, \{p\}), p(s))ds.
    $$


    Перепишем это равенство, подставив $p_0(t) + \Delta p(t)$ вместо $p(t)$:
    \begin{equation}
    \label{sixth_eq}
    x(t, \{p_0 + \Delta p\}) = x_0 + \int\limits_0^t f(s, x(s, \{p_0 + \Delta p\}), (p_0 + \Delta p)(s))ds.
    \end{equation}

    Вычитая из равенства (\ref{sixth_eq}) равенство (\ref{fifth_eq}), получаем равенство

    \begin{equation}
    \label{seventh_eq}
    \Delta x(t) = \int\limits_0^t [f(s, x(s, \{p_0 + \Delta p\}), (p_0 + \Delta p)(s)) - f(s, x(s,\{p_0\}),p_0(s))] ds.
    \end{equation}


    Покажем, что для $\Delta x(t)$ имеет место представление

    $$
    \Delta x(t) = \int\limits_0^t \frac{\partial f}{\partial x}(s, x(s, \{p_0\}),p_0(s)) \cdot \Delta x(s)ds +
    $$
    \begin{equation}
    \label{eighth_eq}
    +\int\limits_0^t \frac{\partial f}{\partial p}(s, x(s, \{p_0\}),p_0(s)) \cdot \Delta p(s)ds + \Gamma(t)
    \end{equation}
    для любого $t \in [0, T]$, причем

    \begin{equation}
    \label{eighth_equ}
    \frac{1}{\|\Delta p\|} \cdot \left|\frac{d\Gamma}{dt}(t)\right| \rightrightarrows 0 \mbox{ при } \|\Delta p\| \rightarrow 0, \mbox{ }t \in [0,T].
    \end{equation}


    Распишем подынтегральную функцию из правой части равенства (\ref{seventh_eq}) в первом приближении:


    $$
    f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), p_0(s)) =
    $$
    $$
    = \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta x(s) +
    $$
    \begin{equation}
    \label{ninth_eq}
    +\frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta p(s) + \alpha(s, \Delta x(s),\Delta p(s)).
    \end{equation}

    Из формулы (\ref{ninth_eq}) следует, что
    $$
    \alpha(s, \Delta x(s),\Delta p(s))= f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) -
    $$
    $$
    - f(s, x(s, \{p_0\}), p_0(s)) -\frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta x(s) -
    $$
    $$
    - \frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta p(s).
    $$

    Подставим результат (\ref{ninth_eq}) в равенство (\ref{seventh_eq}):

    $$
    \Delta x(t) = \int\limits_0^t \frac{\partial f}{\partial x}(s, x(s, \{p_0\}),p_0(s)) \cdot \Delta x(s)ds +
    $$
    \begin{equation}
    \label{tenth_eq}
    + \int\limits_0^t \frac{\partial f}{\partial p}(s, x(s, \{p_0\}),p_0(s)) \cdot \Delta p(s)ds + \int\limits_0^t \alpha(s, \Delta x(s),\Delta p(s)) ds.
    \end{equation}




    Оценим $\alpha(s, \Delta x(s),\Delta p(s))$:

    $$
    |\alpha(s, \Delta x(s),\Delta p(s))| = |f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) -
    $$
    $$
    - f(s, x(s, \{p_0\}), p_0(s)) - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta x(s) -
    $$
    $$
    - \frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta p(s)| \leq
    $$
    $$
    \leq |f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) -
    $$
    $$
    - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s))\cdot\Delta x(s)| +
    $$
    $$
    + \left| \left[ \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s))\right] \cdot \Delta x(s) \right| +
    $$
    $$
    + |f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s,x(s, \{p_0\}),p_0(s))-
    $$
    \begin{equation}
    \label{eleventh_eq}
    - \frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta p(s)|.
    \end{equation}

    Введем обозначение:
    $$
    g(\theta) = f(s, x(s, \{p_0\}) + \theta\cdot\Delta x(s), (p_0 + \Delta p)(s)),\; \theta \in [0,1].
    $$

    С учетом введенного обозначения преобразуем первое выражение в правой части формулы (\ref{eleventh_eq}).

    Верно равенство

    $$
    f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) =
    $$
    $$
    = g(1) - g(0) = \int_0^1\frac{\partial g}{\partial \theta} d \theta = \int_0^1 \frac{\partial f}{\partial x}(s, x(s,\{p_0\})+\theta \Delta x(s),(p_0+\Delta p)(s)) \times
    $$
    $$
    \times \Delta x(s) d \theta.
    $$

    Заметим также, что
    $$
    \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s))\cdot\Delta x(s) =
    $$
    $$
    = \int_0^1 \left [\frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s))\cdot \Delta x(s)\right ] d \theta.
    $$

    Таким образом,
    $$
    f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) -
    $$
    $$
    - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s))\cdot\Delta x(s) =
    $$
    $$
    = \int_0^1 [\frac{\partial f}{\partial x}(s, x(s,\{p_0\}) + \theta \Delta x(s),(p_0+\Delta p)(s)) -
    $$
    $$
    - \frac{\partial f}{\partial x}(s, x(s,\{p_0\}), (p_0+\Delta p)(s))] \cdot \Delta x(s) d \theta.
    $$


    Рассмотрим компактное подмножество $M_a$ пространства $\mathbb R \times \mathbb R^n \times \mathbb R^m$, заданное так:
    $$
    M_a = \{(t,x,p): t \in [0,T], |x - x(t,\{p_0\})| \leq a, |p-p_0(t)| \leq a\},
    $$
    где $a > 0$. Из (\ref{first_equ}) следует, что $a$ можно выбрать столь малым, чтобы выполнялось включение $M_a \in D$.

    \medskip
    Так как $f$ непрерывно дифференцируема по $x$, то $\frac{\partial f}{\partial x}$ равномерно непрерывна на компакте $M_a$. Тогда для любого $\epsilon > 0$ существует такое $\delta \in (0,a)$, что если $|\Delta x(s)|,|\Delta p(s)| \leq \delta$, то
    $$
    |\frac{\partial f}{\partial x}(s, x(s,\{p_0\})+\theta \Delta x(s),(p_0+\Delta p)(s)) - \frac{\partial f}{\partial x}(s, x(s,\{p_0\}), (p_0+\Delta p)(s))| < \epsilon
    $$
    для любых $\theta \in [0,1]$ и $s \in [0,T]$.

    \medskip
    Так как $p$ равномерно стремится к $p_0$ при $\|\Delta p\| \rightarrow 0$, то по теореме о непрерывной зависимости решений от начальных данных и параметров решение $x(t,\{p_0 + \Delta p\})$ равномерно стремится к $x(t,\{p_0\})$ на $[0, T]$.

    \medskip
    Поэтому существует такое $\delta_1 > 0$, что если $\|\Delta p\| < \delta_1$, то выполняются неравенства
    $$
    |x(t, \{p_0 + \Delta p\}) - x(t, \{p_0\})| \leq \delta \mbox{ при } t\in [0,T],
    $$
    то есть точки $(t, x(t, \{p_0 + \Delta p\}), (p_0 + \Delta p)(t))$ и $(t, x(t, p_0), (p_0 + \Delta p)(t))$ лежат в множестве $M_a$ и $|\Delta x(s)| \leq \delta,$ $s\in[0,T] $.

    Тогда

    $$
    \left\|\frac{\partial f}{\partial x}(s, x(s,\{p_0\})+\theta \Delta x(s),(p_0+\Delta p)(s)) - \frac{\partial f}{\partial x}(s, x(s,\{p_0\}), (p_0+\Delta p)(s))\right\| \times
    $$
    $$
    \times |\Delta x (s)| < \epsilon \cdot |\Delta x (s)|.
    $$

    Из этого следует, что если $\|\Delta p\| \leq \delta_1$, то
    $$
    |f(s, x(s, \{p_0\}) + \Delta x(s), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) -
    $$
    \begin{equation}
    \label{twelveth_eq}
    - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s))\cdot\Delta x(s)| \leq \epsilon \cdot \|\Delta x\|.
    \end{equation}

    Теперь аналогично оценим второе выражение в правой части выражения (\ref{eleventh_eq}). Так как $\frac{\partial f}{\partial x}$ равномерно непрерывна на компакте $M_a$, то для любого $\epsilon > 0$ существует такое $\delta \in (0,a)$, что если $\|\Delta p\| \leq \delta$, то
    $$
    \left\|\frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s))\right\| < \epsilon.
    $$
    Из этого следует, что если $\|\Delta p\| \leq \delta$, то верна оценка:

    \begin{equation}
    \label{thirteenth_eq}
    \left|\left[\frac{\partial f}{\partial x} (s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s))\right] \cdot \Delta x(s)\right| \leq \epsilon \cdot \|\Delta x\|.
    \end{equation}

    Аналогичным образом оценим и третье слагаемое в правой части выражения (\ref{eleventh_eq}).

    Введем обозначение:
    $$
    h(\theta) = f(s, x(s, \{p_0\}), (p_0 + \theta\cdot\Delta p)(s)),\; \theta \in [0,1].
    $$

    С учетом введенного обозначения преобразуем третье выражение в правой части формулы (\ref{eleventh_eq}).

    Верно равенство

    $$
    f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), p_0(s)) =
    $$
    $$
    = h(1) - h(0) = \int_0^1\frac{\partial h}{\partial \theta} d \theta = \int_0^1 \frac{\partial f}{\partial p}(s, x(s,\{p_0\}), (p_0 + \theta \cdot \Delta p)(s)) \times
    $$
    $$
    \times \Delta p(s) d \theta.
    $$

    Заметим также, что

    $$
    \frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta p(s) = \int_0^1 \frac{\partial f}{\partial p} (s, x(s,\{p_0\}),p_0(s)) \cdot \Delta p(s) d \theta.
    $$

    Таким образом,
    $$
    f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s,x(s, \{p_0\}),p_0(s))-
    $$
    $$
    -\frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\cdot\Delta p(s) =
    $$
    $$
    = \int_0^1 \frac{\partial f}{\partial p} (s, x(s,\{p_0\}),(p_0 + \theta\cdot\Delta p)(s)) \cdot \Delta p(s) d \theta -
    $$
    $$
    - \int_0^1 \frac{\partial f}{\partial p} (s, x(s,\{p_0\}),p_0(s)) \cdot \Delta p(s) d \theta =
    $$
    $$
    = \int_0^1 \left \{ \left [\frac{\partial f}{\partial p} (s, x(s,\{p_0\}),(p_0 + \theta\cdot\Delta p)(s)) - \frac{\partial f}{\partial p} (s, x(s,\{p_0\}),p_0(s)) \right ] \cdot \Delta p(s) \right \} ds.
    $$

    Так как $f$ непрерывно дифференцируема, то $\frac{\partial f}{\partial p}$ равномерно непрерывна на компакте $M_a$. Тогда для любого $\epsilon > 0$ существует такое $\delta  \in (0,a)$, что если $|\Delta p(s)| \leq \delta$, то

    $$
    \left|\frac{\partial f}{\partial p} (s, x(s,\{p_0\}),(p_0 + \theta\cdot\Delta p)(s)) - \frac{\partial f}{\partial p} (s, x(s,\{p_0\}),p_0(s))\right| < \epsilon.
    $$

    Из этого следует, что если $\|\Delta p\| \leq \delta$, то верна оценка:

    \begin{equation}
    \label{fourteenth_eq}
    \left|\left [\frac{\partial f}{\partial p} (s, x(s,\{p_0\}),p_0 + \theta\cdot\Delta p) - \frac{\partial f}{\partial p} (s, x(s,\{p_0\}),p_0) \right ]\right| \cdot \|\Delta p\| \leq \epsilon \cdot \|\Delta p\|.
    \end{equation}

    Таким образом, из соотношений (\ref{twelveth_eq}) - (\ref{fourteenth_eq}) следует, что для любого $\epsilon > 0$ существует такое $\delta > 0$, что если $\|\Delta p\| < \delta$, то

    $$
    |\alpha(s, \Delta x(s),\Delta p(s))| \leq \epsilon \cdot \|\Delta x\| + \epsilon \cdot \|\Delta x\| + \epsilon \cdot \|\Delta p\|;
    $$
    это означает, что

    $$
    \frac{|\alpha(s, \Delta x(s),\Delta p(s))|}{\|\Delta p\| + \|\Delta x\|} \rightrightarrows 0 \mbox{ при } \|\Delta p\| + \|\Delta x\| \rightarrow 0 \mbox{ }, s \in [0,T].
    $$


    \medskip
    \medskip
    \medskip
    \medskip

    \textbf{Лемма 1.} Верно соотношение
    \begin{equation}
    \label{fourteenth_equ}
    \frac{|\alpha(s, \Delta x(s),\Delta p(s))|}{\|\Delta p\|} \rightrightarrows 0 \mbox{ при } \|\Delta p\| \rightarrow 0 \mbox{ }, s \in [0,T].
    \end{equation}

    \medskip
    \textbf{Доказательство.}

    Оценим сверху функцию, стоящую в левой части равенства (\ref{seventh_eq}):
    $$
    |\Delta x(t)| \leq \int\limits_0^t |f(s, x(s, \{p_0 + \Delta p\}), (p_0 + \Delta p)(s)) -
    $$
    $$
    - f(s, x(s, \{p_0\}), p_0(s))|ds =
    $$
    $$
    = \int\limits_0^t |f(s, x(s, \{p_0 + \Delta p\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) +
    $$
    $$
    + f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), p_0(s))|ds \leq
    $$
    $$
    \leq \int\limits_0^t |f(s, x(s, \{p_0 + \Delta p\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s))|ds +
    $$
    \begin{equation}
    \label{fifteenth_eq}
    + \int\limits_0^t |f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), p_0(s))|ds.
    \end{equation}

    \medskip
    \medskip
    Так как $f$ непрерывно дифференцируема по $x$ и $p$, то она липшицева по этим аргументам на множестве $M_a$. Обозначим соответствующую константу Липшица через $L$.

    Будем считать, как и выше, что величина $\|\Delta p\|$ столь мала, что все аргументы рассматриваемых функций лежат в $M_a$.

    Тогда

    $$
    |f(s, x(s, \{p_0 + \Delta p\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s))| \leq
    $$
    \begin{equation}
    \label{sixteenth_eq}
    \leq L \cdot \|\Delta x\|.
    \end{equation}

    Из того, что $f$ липшицева и по третьему аргументу на множестве $M_a$, следует, что выполняется неравенство
    $$
    |f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), p_0(s))| \leq L \cdot \|\Delta p \|.
    $$

    \medskip
    Из этого следует, что
    $$
    \int\limits_0^t \|f(s, x(s, \{p_0\}), (p_0 + \Delta p)(s)) - f(s, x(s, \{p_0\}), p_0(s))\|ds \leq t \cdot L \cdot \|\Delta p\| \leq
    $$
    \begin{equation}
    \label{seventeenth_eq}
    \leq T \cdot L \cdot \|\Delta p\|, \mbox{ } t \in [0,T].
    \end{equation}

    \medskip
    Подставим оценки (\ref{sixteenth_eq}) и (\ref{seventeenth_eq}) в (\ref{fifteenth_eq}):

    $$
    |\Delta x(t)| \leq L \cdot \int\limits_0^t |\Delta x(s)|ds + T \cdot L \cdot \|\Delta p\|.
    $$

    Применяя к данному неравенству лемму Гронуолла, получаем следующую оценку:
    \begin{equation}
    \label{eighteenth_eq}
    |\Delta x(t)| \leq L \cdot \|\Delta p\| \cdot T \cdot e^{Lt}.
    \end{equation}

    \medskip
    Заметим, что $e^{Lt} \leq e^{LT}$.
    Обозначим $c_1 = L \cdot T \cdot e^{LT}$ и подставим $c_1$ в (\ref{eighteenth_eq}):
    $$
    |\Delta x(t) | \leq c_1 \cdot \|\Delta p\|.
    $$

    \medskip
    Отсюда следует, что существует такое $\delta > 0$, что $\|\Delta x\| \leq c_1 \cdot \|\Delta p\|$ при $\|\Delta p\| < \delta$.

    \medskip
    \medskip
    \medskip
    \medskip

    Таким образом, если величина $\|\Delta p\|$ достаточно мала, то
    $$
    \Delta x(t) = \int\limits_0^t \frac{\partial f}{\partial x}(s, x(s, \{p_0\}),p_0(s)) \cdot \Delta x(s)ds +
    $$
    \begin{equation}
    \label{nineteenth_eq}
    + \int\limits_0^t \frac{\partial f}{\partial p}(s, x(s, \{p_0\}),p_0(s)) \cdot \Delta p(s)ds + \int\limits_0^t \alpha(s, \Delta x(s),\Delta p(s)) ds,
    \end{equation}

    и выполнено соотношение (\ref{fourteenth_equ}).

    Лемма 1 доказана.

    Введем обозначение:

    $$
    \Gamma (t) = \int\limits_0^t \alpha(s, \Delta x(s),\Delta p(s))ds.
    $$

    Из Леммы 1 следует, что $\Delta x(t)$ может быть представлено в виде (\ref{eighth_eq}), и при этом выполнено соотношение (\ref{eighth_equ}).
    \medskip

    Фиксируем вектор $\xi_0 \in \mathbb{R}^n$ и обозначим через $C_{\xi_0}^1[0,T]$ множество таких функций $\xi(t) \in C^1[0,T]$, что $\xi(0) = \xi_0$.

    Рассмотрим оператор
    $$
    \Phi : C_{\xi_0}^1[0,T] \rightarrow C_{\xi_0}^1[0,T],
    $$
    заданный формулой

    \begin{equation}
    \label{twentyfirst_eq}
    \Phi(\xi) = \xi(t) - \int\limits_0^t \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s)) \xi(s)ds,
    \end{equation}
    где $t \in [0, T]$.

    В нашем случае роль функции $\xi(t)$ будет играть $\Delta x(t)$; эта функция принадлежит классу $C^1[0,T]$ как разность решений.

    \medskip
    \textbf{Лемма 2.} При любом $\xi_0$ оператор $\Phi$ обратим.

    \medskip
    \textbf{Доказательство.} Рассмотрим уравнение

    \begin{equation}
    \label{twentysecond_eq}
    \Phi(\xi) = \eta,
    \end{equation}
    где $\eta$ -- произвольная функция, принадлежащая классу $C_{\xi_0}^1[0,T]$.

    Покажем, что такое уравнение единственным образом разрешимо относительно $\xi$ для любого $\eta$. Это и докажет существование оператора $\Phi^{-1}$.

    Продифференцируем равенство (\ref{twentysecond_eq}) по $t$:

    $$
    \dot \xi - X(t) \cdot \xi = \dot \eta,
    $$
    где
    $$
    X(t) = \frac{\partial f}{\partial x} (t, x(t, \{p_0\}), p_0(t)).
    $$

    Таким образом,
    \begin{equation}
    \label{twentysecond_equ}
    \dot \xi = X(t) \cdot \xi + \dot \eta.
    \end{equation}

    Полученная система является линейной неоднородной системой дифференциальных уравнений.
    Соответствующая ей однородная система дифференциальных уравнений имеет вид:

    $$
    \dot \xi = X(t) \cdot \xi.
    $$

    По формуле Лагранжа, учитывая введенное в начале обозначение, решение данной системы с начальными данными $(0, \xi_0)$ имеет вид:

    $$
    \xi(t) = Y(t) \cdot (\xi_0 + \int_0^t Y^{-1} (s) \dot \eta (s) ds).
    $$

    Так как функция $\dot \eta$ непрерывна, то из теоремы о единственности решения для линейных дифференциальных уравнений следует единственность функции $\xi(t)$, для которой выполнено равенство (\ref{twentysecond_equ}), а тем самым и единственность элемента $\xi \in C_{\xi_0}^1[0,1]$, удовлетворяющего равенству (\ref{twentysecond_eq}).

    Таким образом,

    $$
    \Phi^{-1}(\eta) = Y(t) \cdot (\xi_0 + \int_0^t Y^{-1} (s) \dot \eta (s) ds).
    $$

    В дальнейшем мы будем применять обратный функционал $\Phi^{-1}$ при $\xi_0 = 0$. Он задается формулой

    \begin{equation}
    \label{twentythird_eq}
    \Phi^{-1}(\eta) = Y(t) \cdot \int_0^t Y^{-1} (s) \dot \eta (s) ds.
    \end{equation}

    Из формулы (\ref{twentythird_eq}) следует, что существует такое $N>0$, что
    \begin{equation}
    \label{twentyfourth_eq}
    \|\Phi^{-1}(\eta)\| \leq N \cdot \left\|\frac{d\eta}{dt}\right\|.
    \end{equation}

    Рассмотрим $\xi(t) = \Delta x(t)$ и запишем формулу (\ref{eighth_eq}) в виде

    $$
    \Phi(\xi) = \xi(t) - \int\limits_0^t \frac{\partial f}{\partial x} (s, x(s, \{p_0\}), p_0(s)) \xi(s)ds =
    $$
    \begin{equation}
    \label{twentyfifth_eq}
    = \int_0^t \frac{\partial f}{\partial p}(s, x(s, \{p_0\}),(p_0)(s)) \cdot \Delta p(s)ds + \Gamma(t).
    \end{equation}

    Применим к равенству (\ref{twentyfifth_eq}) оператор $\Phi^{-1}$ (напомним, что мы рассматриваем $\xi_0 = 0$):

    \begin{equation}
    \label{twentysixth_eq}
    \xi(t) = Y(t)\cdot \int\limits_0^t Y^{-1} (s) \cdot \frac{\partial f}{\partial p} (s, x(s, \{p_0\}), p_0(s))\Delta p(s)ds + \Phi^{-1}(\Gamma)(t).
    \end{equation}

    Подставим в (\ref{twentysixth_eq}) $t=T$ (учитывая вид функционала $\Psi$, введенного в (\ref{third_eq})):

    \begin{equation}
    \label{twentyseventh_eq}
    \xi(T) = Y(T)\cdot \Psi(\Delta p) + G(\Delta p),
    \end{equation}
    где
    $$
    G(\Delta p) = \Phi^{-1}(\Gamma)(T).
    $$

    Из соотношения (\ref{eighth_equ}) и оценки (\ref{twentyfourth_eq}) следует, что

    \begin{equation}
    \label{twentyeighth_eq}
    \frac{|G(\Delta p)|}{\|\Delta p\|} \rightarrow 0 \mbox{ при } \|\Delta p\| \rightarrow 0.
    \end{equation}

    Перейдем к доказательству основной теоремы.

    Предположим противное: пусть существует такая последовательность $p_k \in \EuScript P$, что $p_k \not\equiv p_0$, $\|p_k - p_0\|\rightarrow 0$ при $k \rightarrow \infty$ и $x(T, \{p_0\}) = x(T, \{p_k\})$.

    Так как $\|p_k - p_0\| \rightarrow 0$, то приведенные выше рассуждения применимы к функциям

    $$
    \Delta_k x(t) = x(t, \{p_k\}) - x(t, \{p_0\})
    $$
    при достаточно больших $k$.

    Тогда при достаточно больших $k$ выполняются равенства:

    \begin{equation}
    \label{twentyninth_eq}
    0 = Y(T)\cdot \Psi(\Delta p_k) + G(\Delta p_k),
    \end{equation}
    где $\Delta p_k = p_k(t) - p_0(t)$.

    Разделим равенство (\ref{twentyninth_eq}) на $\|\Delta p_k\|$:

    \begin{equation}
    \label{thirtyth_eq}
    0 = Y(T)\cdot \Psi\left(\frac{\Delta p_k}{\|\Delta p_k\|}\right) + \frac{G(\Delta p_k)}{\|\Delta p_k\|}
    \end{equation}

    По предположению, семейство $\EuScript P$ нормированно отделено от $\EuScript K$ в $p_0$, поэтому существует такая подпоследовательность $p_{k_l}$ последовательности $p_k$, что

    \begin{equation}
    \label{thirtyfirst_eq}
    \frac{\Delta p_{k_l}(t)}{\|\Delta p_{k_l}\|}\rightrightarrows r \mbox{ при } k_l\rightarrow \infty, t \in [0,T],
    \end{equation}
    и $r \not\in \EuScript K$.

    Так как функционал $\Psi$ непрерывен, из (\ref{thirtyfirst_eq}) следует, что
    \begin{equation}
    \label{thirtysecond_eq}
    \Psi\left(\frac{\Delta p_{k_l}(t)}{\|\Delta p_{k_l}\|}\right)\rightarrow\Psi(r) \neq 0 \mbox{ при } k_l\rightarrow \infty, t \in [0,T].
    \end{equation}

    Матрица $Y(T)$ невырождена (как фундаментальная матрица), поэтому

    \begin{equation}
    \label{thirtythird_eq}
    Y(T) \cdot \Psi(r) \neq 0.
    \end{equation}

    С другой стороны, из оценки (\ref{twentyeighth_eq}) следует, что
    \begin{equation}
    \label{thirtyfourth_eq}
    \frac{|G(\Delta p_{k_l})|}{\|\Delta p_{k_l}\|} \rightarrow 0 \mbox{ при } k_l \rightarrow \infty.
    \end{equation}

    Из (\ref{thirtyth_eq}) и (\ref{thirtyfourth_eq}) следует соотношение
    $$
    0 = Y(T)\cdot\Psi(r) + 0,
    $$
    противоречащее соотношению (\ref{thirtythird_eq}).

    Теорема доказана.
